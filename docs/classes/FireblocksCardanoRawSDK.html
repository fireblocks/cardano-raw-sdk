      Returns Promise<StakingTransactionResult & { rewardAmount?: number }>

      Throws

      Error if withdrawal fails

      Example

      // Withdraw all available rewards
      const result = await sdk.withdrawRewards({
        vaultAccountId: "0",
        fee: 300000 // 0.3 ADA
      });

      // Withdraw up to 5 ADA
      const result = await sdk.withdrawRewards({
        vaultAccountId: "0",
        limit: 5000000, // 5 ADA in Lovelace

      });
